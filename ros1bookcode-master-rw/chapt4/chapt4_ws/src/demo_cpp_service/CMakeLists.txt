cmake_minimum_required(VERSION 3.0.2)
project(demo_cpp_service)
find_package(catkin REQUIRED COMPONENTS roscpp geometry_msgs turtlesim chapt4_interfaces)
catkin_package(CATKIN_DEPENDS roscpp geometry_msgs turtlesim chapt4_interfaces)
include_directories(${catkin_INCLUDE_DIRS})
add_executable(patrol_client src/patrol_client.cpp)
add_executable(turtle_control src/turtle_control.cpp)
foreach(t patrol_client turtle_control)
  add_dependencies(${t} ${${PROJECT_NAME}_EXPORTED_TARGETS} ${catkin_EXPORTED_TARGETS})
  target_link_libraries(${t} ${catkin_LIBRARIES})
endforeach()
install(TARGETS patrol_client turtle_control RUNTIME DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION})
install(DIRECTORY launch DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION})
