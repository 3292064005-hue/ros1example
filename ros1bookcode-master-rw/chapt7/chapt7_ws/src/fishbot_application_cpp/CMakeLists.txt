cmake_minimum_required(VERSION 3.0.2)
project(fishbot_application_cpp)
find_package(catkin REQUIRED COMPONENTS roscpp actionlib move_base_msgs tf)
catkin_package(CATKIN_DEPENDS roscpp actionlib move_base_msgs tf)
include_directories(${catkin_INCLUDE_DIRS} include)
add_executable(nav_to_pose_cpp src/nav_to_pose.cpp)
add_dependencies(nav_to_pose_cpp ${catkin_EXPORTED_TARGETS})
target_link_libraries(nav_to_pose_cpp ${catkin_LIBRARIES})
install(TARGETS nav_to_pose_cpp RUNTIME DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION})
