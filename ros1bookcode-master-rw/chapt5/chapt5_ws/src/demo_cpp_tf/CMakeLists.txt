cmake_minimum_required(VERSION 3.0.2)
project(demo_cpp_tf)
find_package(catkin REQUIRED COMPONENTS roscpp geometry_msgs tf2 tf2_ros tf2_geometry_msgs)
catkin_package(CATKIN_DEPENDS roscpp geometry_msgs tf2 tf2_ros tf2_geometry_msgs)
include_directories(${catkin_INCLUDE_DIRS})
add_executable(dynamic_tf_broadcaster src/dynamic_tf_broadcaster.cpp)
add_executable(static_tf_broadcaster src/static_tf_broadcaster.cpp)
add_executable(tf_listener src/tf_listener.cpp)
foreach(t dynamic_tf_broadcaster static_tf_broadcaster tf_listener)
  target_link_libraries(${t} ${catkin_LIBRARIES})
endforeach()
install(TARGETS dynamic_tf_broadcaster static_tf_broadcaster tf_listener RUNTIME DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION})
