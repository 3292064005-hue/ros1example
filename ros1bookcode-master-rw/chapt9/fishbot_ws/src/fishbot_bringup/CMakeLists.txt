cmake_minimum_required(VERSION 3.0.2)
project(fishbot_bringup)
find_package(catkin REQUIRED COMPONENTS roscpp geometry_msgs nav_msgs tf2_ros)
catkin_package(CATKIN_DEPENDS roscpp geometry_msgs nav_msgs tf2_ros)
include_directories(${catkin_INCLUDE_DIRS} include)
add_executable(odom2tf src/odom2tf.cpp)
add_executable(mock_odom_publisher src/mock_odom_publisher.cpp)
add_dependencies(odom2tf ${catkin_EXPORTED_TARGETS})
add_dependencies(mock_odom_publisher ${catkin_EXPORTED_TARGETS})
target_link_libraries(odom2tf ${catkin_LIBRARIES})
target_link_libraries(mock_odom_publisher ${catkin_LIBRARIES})
install(DIRECTORY launch DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION})
install(DIRECTORY map DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION})
install(TARGETS odom2tf mock_odom_publisher RUNTIME DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION})
